Project: Create a neural network class¶

Based on previous code examples, develop a neural network class that is able to classify any dataset provided. The class should create objects based on the desired network architecture:

  1. Number of inputs
  2. Number of hidden layers
  3. Number of neurons per layer
  4. Number of outputs
  5. Learning rate

The class must have the train, and predict functions.

Test the neural network class on the datasets provided below: Use the input data to train the network, and then pass new inputs to predict on. Print the expected label and the predicted label for the input you used. Print the accuracy of the training after predicting on different inputs.

Use matplotlib to plot the error that the train method generates.

Don't forget to install Keras and tensorflow in your environment!

class NeuralNetwork:

    def __init__(self, ninputs, nlayers, nneurons, noutputs, alpha):
        '''
            layers: List of integers which represents the architecture of the network.
            alpha:  Learning rate.
        '''
        
        # TODO: Initialize the list of weights matrices, then store
        # the network architecture and learning rate
    
             
        self.layers = nlayers
        self.neurons = nneurons
        self.alpha = alpha
        
        self.weighti = np.random.randn(ninputs, nneurons)
        self.weights = np.zeros((nlayers-1, nneurons, nneurons))
        self.weighto = np.random.randn(nneurons,noutputs)
        
        for i in range(nlayers-1):
            self.weights[i] = np.random.randn(nneurons,nneurons)
            
      
        
        self.bi = np.random.randn(nneurons)  
        self.bs = np.random.randn(nlayers-1, nneurons)
        self.bout = np.random.randn(noutputs)
    
            
    def __repr__(self): 
        # construct and return a string that represents the network 
        # architecture 
        return "NeuralNetwork: {}".format( "-".join(str(l) for l in self.layers))

    def softmax(self,X):  
        # applies the softmax function to a set of values
        
        expX = np.exp(X)
        return expX / expX.sum(axis=1, keepdims=True)
        
    def sigmoid(self, x):
        # the sigmoid for a given input value
        
        return 1.0 / (1.0 + np.exp(-x))
    
    def sigmoid_deriv(self, x):
        # the derivative of the sigmoid
        
        return x * (1 - x)
    
    def predict(self, inputs):
        level0 = inputs
        levels = np.zeros((self.layers,inputs.shape[0],self.neurons))
        levels[0] = self.sigmoid(np.dot(level0,self.weighti)+self.bi)
        
        for i in range(1 , self.layers):
            levels[i] = self.sigmoid(np.dot(levels[i-1],self.weights[i-1])+self.bs[i-1])
            
        pred = self.softmax(np.dot(levels[len(levels)-1], self.weighto)+self.bout)

        return pred,levels

    def train(self, inputs, labels, epochs = 1000, displayUpdate = 100):

        # TODO: Define the training step for the network. It should include the forward and back propagation
        # steps, the updating of the weights, and it should print the error every 'displayUpdate' epochs
        # It must return the errors so that they can be displayed with matplotlib

        errors = []

        for epoch in range(epochs):


            pred, levels= self.predict(inputs)

            level0 = inputs
            level1 = levels[len(levels)-1]
            level2 = pred

            level_error = labels-level2


            if epoch%displayUpdate == 0:
                print("Error:", np.mean(np.abs(level_error)))

            errors.append(np.average(np.abs(level_error)))

            level_delta = level_error * self.sigmoid_deriv(pred)

            b_delta = np.sum(level_delta)
            self.bout += b_delta * self.alpha

            level_error = np.dot(level_delta, self.weighto.T)
            self.weighto += np.dot(level1.T,level_delta)*self.alpha 
            level_delta = level_error * self.sigmoid_deriv(level1)

            for i in range(self.layers-1):

                b_delta = np.sum(level_delta)
                self.bs[len(self.bs)-i-1] += b_delta * self.alpha 

                level_error = np.dot(level_delta, self.weights[len(self.weights)-i-1])
                self.weights[len(self.weights)-i-1] += np.dot(levels[len(levels)-i-2].T,level_delta)*self.alpha
                level_delta = level_error * self.sigmoid_deriv(levels[len(levels)-i-2])


            b_delta = np.sum(level_delta)
            self.bi+= b_delta * self.alpha

            self.weighti += np.dot(level0.T,level_delta)*self.alpha
        return errors
